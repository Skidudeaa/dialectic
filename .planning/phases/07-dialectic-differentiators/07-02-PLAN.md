Create a custom hook for forking threads using React Query mutation pattern.

```typescript
// hooks/use-fork.ts
import { useMutation, useQueryClient } from '@tanstack/react-query';
import { api } from '@/lib/api';
import { useRouter } from 'expo-router';
import { Alert } from 'react-native';

interface ForkParams {
  roomId: string;
  sourceThreadId: string;
  forkAfterMessageId: string;
  title?: string;
}

interface ThreadResponse {
  id: string;
  room_id: string;
  parent_thread_id: string;
  title: string | null;
  message_count: number;
}

export function useForkThread() {
  const queryClient = useQueryClient();
  const router = useRouter();

  const mutation = useMutation({
    mutationFn: async (params: ForkParams): Promise<ThreadResponse> => {
      const response = await api.post(
        `/threads/${params.sourceThreadId}/fork`,
        {
          source_thread_id: params.sourceThreadId,
          fork_after_message_id: params.forkAfterMessageId,
          title: params.title,
        }
      );
      return response.data;
    },

    onSuccess: (newThread, params) => {
      // Invalidate threads list
      queryClient.invalidateQueries({
        queryKey: ['threads', params.roomId],
      });

      // CONTEXT.md: After forking, navigate immediately to new thread
      router.push(`/room/${params.roomId}/thread/${newThread.id}`);
    },

    onError: (error) => {
      Alert.alert('Fork Failed', 'Could not create branch. Please try again.');
      console.error('Fork error:', error);
    },
  });

  return {
    forkThread: mutation.mutateAsync,
    isForking: mutation.isPending,
    error: mutation.error,
  };
}
```

The hook should:
1. Call POST /threads/{id}/fork
2. Invalidate threads query on success
3. Navigate to new thread using expo-router
4. Handle errors with user-friendly alert

```typescript
// components/chat/message-context-menu.tsx
import React, { useCallback } from 'react';
import { HoldItem } from 'react-native-hold-menu';
import { Alert } from 'react-native';
import * as Clipboard from 'expo-clipboard';
import { useForkThread } from '@/hooks/use-fork';

interface MessageContextMenuProps {
  children: React.ReactNode;
  messageId: string;
  messageContent: string;
  threadId: string;
  roomId: string;
}

export function MessageContextMenu({
  children,
  messageId,
  messageContent,
  threadId,
  roomId,
}: MessageContextMenuProps) {
  const { forkThread, isForking } = useForkThread();

  const handleFork = useCallback(async () => {
    // CONTEXT.md: Optional naming prompt - allow skip
    Alert.prompt(
      'Name this branch',
      'Give your branch a name (optional)',
      [
        {
          text: 'Skip',
          style: 'cancel',
          onPress: () => {
            forkThread({
              roomId,
              sourceThreadId: threadId,
              forkAfterMessageId: messageId,
              // Auto-generate title from timestamp if skipped
              title: `Branch ${new Date().toLocaleDateString()}`,
            });
          },
        },
        {
          text: 'Create',
          onPress: (title?: string) => {
            forkThread({
              roomId,
              sourceThreadId: threadId,
              forkAfterMessageId: messageId,
              title: title || `Branch ${new Date().toLocaleDateString()}`,
            });
          },
        },
      ],
      'plain-text',
      ''
    );
  }, [roomId, threadId, messageId, forkThread]);

  const handleCopy = useCallback(async () => {
    await Clipboard.setStringAsync(messageContent);
  }, [messageContent]);

  const menuItems = [
    {
      text: 'Fork from here',
      icon: 'git-branch-outline',  // Ionicons name
      onPress: handleFork,
      isDestructive: false,
    },
    {
      text: 'Copy',
      icon: 'copy-outline',
      onPress: handleCopy,
      isDestructive: false,
    },
  ];

  return (
    <HoldItem
      items={menuItems}
      hapticFeedback="Medium"
      activateOn="hold"
      menuAnchorPosition="bottom-center"
      closeOnTap
    >
      {children}
    </HoldItem>
  );
}
```

Note: Alert.prompt is iOS-only. For Android, create a simple modal or use a library. For MVP, the iOS Alert.prompt works, and Android can skip the prompt and auto-generate title.
