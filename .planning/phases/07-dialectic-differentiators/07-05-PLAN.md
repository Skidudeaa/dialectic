import { create } from 'zustand';
import { websocketService } from '@/services/websocket/websocket-service';

interface LLMStreamState {
  threadId: string | null;
  messageId: string | null;
  isThinking: boolean;
  isStreaming: boolean;
  partialContent: string;
  // NEW: Interjection metadata
  speakerType: 'llm_primary' | 'llm_provoker' | null;
  interjectionType: 'summoned' | 'proactive' | null;
  interjectionReason: string | null;
}

interface LLMState {
  // Per-thread streaming state
  streams: Map<string, LLMStreamState>;

  // Actions
  startThinking: (threadId: string) => void;
  startStreaming: (threadId: string, messageId: string) => void;
  appendToken: (threadId: string, token: string) => void;
  completeStream: (threadId: string, content: string, metadata: {
    speakerType: 'llm_primary' | 'llm_provoker';
    interjectionType: 'summoned' | 'proactive';
    interjectionReason?: string;
  }) => void;
  cancelStream: (threadId: string) => void;
  clearStream: (threadId: string) => void;
  getStream: (threadId: string) => LLMStreamState | undefined;
}

const createEmptyStream = (threadId: string): LLMStreamState => ({
  threadId,
  messageId: null,
  isThinking: false,
  isStreaming: false,
  partialContent: '',
  speakerType: null,
  interjectionType: null,
  interjectionReason: null,
});

export const useLLMStore = create<LLMState>((set, get) => ({
  streams: new Map(),

  startThinking: (threadId) => {
    set((state) => {
      const streams = new Map(state.streams);
      streams.set(threadId, {
        ...createEmptyStream(threadId),
        isThinking: true,
      });
      return { streams };
    });
  },

  startStreaming: (threadId, messageId) => {
    set((state) => {
      const streams = new Map(state.streams);
      const current = streams.get(threadId) || createEmptyStream(threadId);
      streams.set(threadId, {
        ...current,
        messageId,
        isThinking: false,
        isStreaming: true,
      });
      return { streams };
    });
  },

  appendToken: (threadId, token) => {
    set((state) => {
      const streams = new Map(state.streams);
      const current = streams.get(threadId);
      if (current) {
        streams.set(threadId, {
          ...current,
          partialContent: current.partialContent + token,
        });
      }
      return { streams };
    });
  },

  completeStream: (threadId, content, metadata) => {
    set((state) => {
      const streams = new Map(state.streams);
      const current = streams.get(threadId);
      if (current) {
        streams.set(threadId, {
          ...current,
          isStreaming: false,
          partialContent: content,
          speakerType: metadata.speakerType,
          interjectionType: metadata.interjectionType,
          interjectionReason: metadata.interjectionReason || null,
        });
      }
      return { streams };
    });
  },

  cancelStream: (threadId) => {
    // Send cancel to server
    websocketService.send({
      type: 'cancel_llm',
      payload: { thread_id: threadId },
    });

    // Clear local state immediately for responsive UI
    set((state) => {
      const streams = new Map(state.streams);
      streams.delete(threadId);
      return { streams };
    });
  },

  clearStream: (threadId) => {
    set((state) => {
      const streams = new Map(state.streams);
      streams.delete(threadId);
      return { streams };
    });
  },

  getStream: (threadId) => get().streams.get(threadId),
}));
```

- Tracks speakerType and interjectionType per stream

- State per-thread enables multi-thread support

// hooks/use-llm.ts
import { useCallback, useEffect } from 'react';
import { useLLMStore } from '@/stores/llm-store';
import { useWebSocket } from './use-websocket';
import type { InboundMessage } from '@/services/websocket/types';

export function useLLM(threadId: string) {
  const stream = useLLMStore((s) => s.getStream(threadId));
  const startThinking = useLLMStore((s) => s.startThinking);
  const startStreaming = useLLMStore((s) => s.startStreaming);
  const appendToken = useLLMStore((s) => s.appendToken);
  const completeStream = useLLMStore((s) => s.completeStream);
  const cancelStream = useLLMStore((s) => s.cancelStream);
  const clearStream = useLLMStore((s) => s.clearStream);

  // Handle LLM WebSocket events
  const handleMessage = useCallback(
    (message: InboundMessage) => {
      const payload = message.payload as Record<string, unknown>;
      const msgThreadId = payload.thread_id as string;

      // Only process events for this thread
      if (msgThreadId !== threadId) return;

      switch (message.type) {
        case 'llm_thinking':
          startThinking(threadId);
          break;

        case 'llm_streaming':
          if (!stream?.isStreaming) {
            startStreaming(threadId, payload.message_id as string);
          }
          appendToken(threadId, payload.token as string);
          break;

        case 'llm_done':
          completeStream(threadId, payload.content as string, {
            speakerType: (payload.speaker_type as 'llm_primary' | 'llm_provoker') || 'llm_primary',
            interjectionType: (payload.interjection_type as 'summoned' | 'proactive') || 'summoned',
            interjectionReason: payload.interjection_reason as string | undefined,
          });
          break;

        case 'llm_error':
        case 'llm_cancelled':
          clearStream(threadId);
          break;
      }
    },
    [threadId, stream?.isStreaming, startThinking, startStreaming, appendToken, completeStream, clearStream]
  );

  // Register handler with WebSocket
  const { addMessageHandler, removeMessageHandler } = useWebSocket();

  useEffect(() => {
    addMessageHandler('llm', handleMessage);
    return () => removeMessageHandler('llm');
  }, [addMessageHandler, removeMessageHandler, handleMessage]);

  // Cancel action
  const handleCancel = useCallback(() => {
    cancelStream(threadId);
  }, [cancelStream, threadId]);

  return {
    isThinking: stream?.isThinking ?? false,
    isStreaming: stream?.isStreaming ?? false,
    partialContent: stream?.partialContent ?? '',
    speakerType: stream?.speakerType ?? null,
    interjectionType: stream?.interjectionType ?? null,
    interjectionReason: stream?.interjectionReason ?? null,
    cancelStream: handleCancel,
  };
}
```

import { useLLM } from '@/hooks/use-llm';

export function MessageList({
  threadId,
  roomId,
  messages,
  // ... other props
}: MessageListProps) {
  // Get LLM stream state for this thread
  const { isThinking, isStreaming, partialContent, speakerType, interjectionType, cancelStream } = useLLM(threadId);

  const renderItem = useCallback(
    ({ item, index }: ListRenderItemInfo<Message>) => {
      const isLLM =
        item.senderId === 'llm' ||
        ['LLM_PRIMARY', 'LLM_PROVOKER'].includes(item.speakerType || '');

      if (isLLM) {
        // Check if this is the active streaming message
        const isActiveStream = isStreaming && index === messages.length - 1;

        return (
          <MessageContextMenu
            messageId={item.id}
            messageContent={item.content}
            threadId={threadId}
            roomId={roomId}
          >
            <LLMMessageBubble
              content={isActiveStream ? partialContent : item.content}
              createdAt={item.createdAt}
              isThinking={isActiveStream && isThinking}
              isStreaming={isActiveStream && isStreaming}
              speakerType={isActiveStream ? speakerType : (item.speakerType?.toLowerCase() as 'llm_primary' | 'llm_provoker')}
              interjectionType={isActiveStream ? interjectionType : undefined}
              onStop={isActiveStream ? cancelStream : undefined}
            />
          </MessageContextMenu>
        );

      return (
        <MessageContextMenu
          messageId={item.id}
          messageContent={item.content}
          threadId={threadId}
          roomId={roomId}
        >
          <MessageBubble message={item} />
        </MessageContextMenu>
      );
    },
    [threadId, roomId, isThinking, isStreaming, partialContent, speakerType, interjectionType, cancelStream, messages.length]
  );

  // ... rest of component
}
