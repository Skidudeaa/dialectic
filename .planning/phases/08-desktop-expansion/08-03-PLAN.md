```bash
mkdir -p packages/windows
```

Create package.json:
```json
{
  "name": "@dialectic/windows",
  "version": "1.0.0",
  "private": true,
  "scripts": {
    "start": "react-native start",
    "windows": "react-native run-windows",
    "build": "react-native run-windows --release"
  },
  "dependencies": {
    "@dialectic/app": "workspace:*",
    "react": "19.1.0",
    "react-native": "0.81.5",
    "react-native-windows": "^0.81.0"
  },
  "devDependencies": {
    "@react-native/babel-preset": "^0.81.0",
    "@react-native/metro-config": "^0.81.0",
    "babel-plugin-module-resolver": "^5.0.0",
    "typescript": "~5.9.2"
  }
}
```

Note: react-native-windows 0.81 aligns with RN 0.81 (mobile version).

Create tsconfig.json:
```json
{
  "extends": "../../tsconfig.base.json",
  "compilerOptions": {
    "baseUrl": ".",
    "paths": {
      "@dialectic/app": ["../app/src"],
      "@dialectic/app/*": ["../app/src/*"]
    }
  },
  "include": ["*.ts", "*.tsx", "src/**/*.ts", "src/**/*.tsx"]
}
```

Create index.js:
```javascript
import { AppRegistry } from 'react-native';
import App from './App';
import { name as appName } from './app.json';

AppRegistry.registerComponent(appName, () => App);
```

```typescript
import React from 'react';
import { View, Text, StyleSheet } from 'react-native';

export default function App() {
  return (
    <View style={styles.container}>
      <Text style={styles.text}>Dialectic for Windows</Text>
    </View>
  );
}

const styles = StyleSheet.create({
  container: { flex: 1, justifyContent: 'center', alignItems: 'center', backgroundColor: '#fff' },
  text: { fontSize: 24, color: '#000' },
});
```

Create packages/windows/metro.config.js for monorepo resolution:

```javascript
const path = require('path');
const { getDefaultConfig, mergeConfig } = require('@react-native/metro-config');

const projectRoot = __dirname;
const workspaceRoot = path.resolve(projectRoot, '../..');

const config = {
  watchFolders: [workspaceRoot],
  resolver: {
    nodeModulesPaths: [
      path.resolve(projectRoot, 'node_modules'),
      path.resolve(workspaceRoot, 'node_modules'),
    ],
    extraNodeModules: {
      '@dialectic/app': path.resolve(workspaceRoot, 'packages/app'),
    },
    // Windows-specific: block packages that don't work on Windows
    blockList: [
      /packages\/mobile\/.*/,
      /packages\/macos\/.*/,
    ],
  },
};

module.exports = mergeConfig(getDefaultConfig(projectRoot), config);
```

Create packages/windows/babel.config.js:
```javascript
module.exports = {
  presets: ['module:@react-native/babel-preset'],
  plugins: [
    ['module-resolver', {
      root: ['./'],
      alias: {
        '@dialectic/app': '../app/src',
      },
    }],
  ],
};
```

Run yarn install to ensure all dependencies resolved:
```bash
cd packages/windows
yarn install
```

Test Metro starts:
```bash
yarn start
```
